CREATE MATERIALIZED VIEW state_yearly_rain AS  SELECT country_code FROM raw_weather_data WHERE state_code IS NOT NULL and one_hour_precip IS NOT NULL AND year IS NOT NULL and day IS NOT NULL AND month IS NOT NULL AND hour IS NOT NULL PRIMARY KEY ((state_code, year), one_hour_precip, wsid, day, month, hour);
